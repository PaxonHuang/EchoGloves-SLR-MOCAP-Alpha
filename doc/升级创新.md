```

## 硬件选择建议

### 方案对比

````
┌──────────────┬─────────────┬─────────┬──────────┬────────────┐
│ 芯片          │ RAM/Flash   │ 性能    │ 功耗     │ 推荐指数   │
├──────────────┼─────────────┼─────────┼──────────┼────────────┤
│ ESP32        │ 8MB/4MB     │ 中等    │ 低       │ ⭐⭐⭐⭐ │
│ ESP32-S3     │ 8MB/8MB     │ 高      │ 低       │ ⭐⭐⭐⭐⭐│
│ STM32H743    │ 864KB/2MB   │ 极高    │ 中       │ ⭐⭐⭐    │
│ nRF52840     │ 256KB/1MB   │ 高      │ 极低     │ ⭐⭐⭐    │
└──────────────┴─────────────┴─────────┴──────────┴────────────┘

推荐: ESP32-S3 (保持现有生态，内存充足)
```

### 不建议用ESP IDF或STM32的原因

````
❌ ESP IDF (裸机)
  - 学习陡峭，开发周期长
  - TensorFlow Lite支持差
  - 不值得从Arduino迁移

❌ STM32
  - 生态不如ESP32
  - TinyML支持弱
  - 功耗优势不明显

✅ 保持ESP32/ESP32-S3 (Arduino框架)
  - TensorFlow Lite for Microcontrollers完美支持
  - Arduino库生态完善
  - OTA更新方便
  - 社区活跃
````

### Phase 1: 数据采集改进 (1-2周)

````cpp
// firmware/src/sensor_fusion.h
struct SensorFrame {
    uint32_t timestamp;
    float flex[5];      // 柔性传感器
    float gyro[3];      // 陀螺仪
    float accel[3];     // 加速度计
};

class RingBuffer {
    static const int BUFFER_SIZE = 100;  // 2秒@50Hz
    SensorFrame buffer[BUFFER_SIZE];
    int head = 0;
    
public:
    void push(const SensorFrame& frame) {
        buffer[head] = frame;
        head = (head + 1) % BUFFER_SIZE;
    }
    
    bool isFull() { return head == 0; }
    
    // 转换为神经网络输入 (100, 11)
    float** getFeatureMatrix() { /*...*/ }
};
````

### Phase 2: 模型转换与优化 (2-3周)

````python
# train/convert_to_tflite.py
import tensorflow as tf
import numpy as np

# 1. 构建模型
model = tf.keras.Sequential([
    tf.keras.layers.Conv1D(32, kernel_size=3, activation='relu', 
                          input_shape=(100, 11)),
    tf.keras.layers.MaxPooling1D(2),
    tf.keras.layers.Conv1D(64, kernel_size=3, activation='relu'),
    tf.keras.layers.MaxPooling1D(2),
    tf.keras.layers.Conv1D(128, kernel_size=3, activation='relu'),
    
    tf.keras.layers.Bidirectional(
        tf.keras.layers.LSTM(256, return_sequences=False)
    ),
    tf.keras.layers.Dropout(0.3),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dense(22, activation='softmax')  # 22种手势
])

model.summary()

# 2. 加载数据
X_train = np.load('train_sequences.npy')  # (N, 100, 11)
y_train = np.load('train_labels.npy')

# 3. 训练
model.compile(optimizer='adam', 
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
model.fit(X_train, y_train, epochs=50, batch_size=32, 
          validation_split=0.2)

# 4. 量化转换
converter = tf.lite.TFLiteConverter.from_keras_model(model)
converter.target_spec.supported_ops = [
    tf.lite.OpsSet.TFLITE_BUILTINS
]
# 动态范围量化
converter.optimizations = [tf.lite.Optimize.DEFAULT]
tflite_model = converter.convert()

with open('gesture_model.tflite', 'wb') as f:
    f.write(tflite_model)

print(f"模型大小: {len(tflite_model)/1024:.1f}KB")  # 应<300KB
````

### Phase 3: ESP32部署 (2周)

````cpp
// firmware/src/main.cpp
#include <TensorFlowLite_ESP32.h>
#include "gesture_model.h"
#include "sensor_fusion.h"

namespace {
    const tflite::Model* model = nullptr;
    tflite::MicroInterpreter* interpreter = nullptr;
    TfLiteTensor* input = nullptr;
    TfLiteTensor* output = nullptr;
    
    // Arena for interpreter (~30KB)
    constexpr int kTensorArenaSize = 30 * 1024;
    alignas(16) uint8_t tensor_arena[kTensorArenaSize];
}

void setup() {
    Serial.begin(115200);
    
    // 加载模型
    model = tflite::GetModel(gesture_model);
    static tflite::MicroMutableOpResolver<10> resolver;
    resolver.AddConv2D();
    resolver.AddMaxPool2D();
    resolver.AddLSTM();
    resolver.AddFullyConnected();
    resolver.AddSoftmax();
    
    static tflite::MicroInterpreter static_interpreter(
        model, resolver, tensor_arena, kTensorArenaSize);
    interpreter = &static_interpreter;
    
    interpreter->AllocateTensors();
    input = interpreter->input(0);
    output = interpreter->output(0);
}

RingBuffer sensor_buffer;

void loop() {
    // 采集传感器数据
    SensorFrame frame = readSensors();
    sensor_buffer.push(frame);
    
    if (sensor_buffer.isFull()) {
        // 获取特征矩阵
        float** features = sensor_buffer.getFeatureMatrix();
        
        // 复制到输入张量
        memcpy(input->data.f, (float*)features, 
               100 * 11 * sizeof(float));
        
        // 推理
        interpreter->Invoke();
        
        // 获取输出
        float* predictions = output->data.f;
        int predicted_class = 0;
        float max_prob = predictions[0];
        
        for (int i = 1; i < 22; i++) {
            if (predictions[i] > max_prob) {
                max_prob = predictions[i];
                predicted_class = i;
            }
        }
        
        Serial.printf("Gesture: %s (%.2f%%)\n", 
                     GESTURE_NAMES[predicted_class],
                     max_prob * 100);
    }
    
    delay(20);  // 50Hz采样
}
````

## 边缘计算的必要性

### ✅ 必要，理由如下

````
1. 延迟要求
   当前: 树莓派方案 100-200ms延迟 ❌ (用户体验差)
   改进: ESP32本地 <50ms延迟 ✅ (实时反馈)

2. 隐私保护
   ✗ 上传原始传感器数据到云端 (安全风险)
   ✓ 本地处理只传结果 (隐私友好)

3. 网络独立性
   ✗ 依赖WiFi (不稳定，功耗高)
   ✓ 离线工作 (可用性强)

4. 功耗优化
   ✗ 持续WiFi通信 (续航<4小时)
   ✓ 仅计算处理 (续航>12小时)
````

### 边缘计算 vs 云端方案

````
┌──────────────┬────────────────┬──────────────────┐
│ 指标          │ 本地推理(推荐) │ 云端推理(不推荐) │
├──────────────┼────────────────┼──────────────────┤
│ 延迟         │ <50ms          │ 100-500ms        │
│ 隐私         │ 完全本地       │ 数据上传         │
│ 离线工作     │ 支持           │ 不支持           │
│ 功耗         │ 低             │ 高               │
│ 设备成本     │ 中等           │ 低               │
│ 运维成本     │ 低             │ 高               │
└──────────────┴────────────────┴──────────────────┘
````

## 完整软硬件升级方案总结

### 硬件层
```
当前:  ESP32 (4MB Flash) + MPU6050
升级:  ESP32-S3 (8MB Flash) + MPU6050 + IMU/AHRS融合
      或者添加压力传感器阵列(手掌压力)
```

### 固件层
```
Arduino → PlatformIO + TensorFlow Lite Micro
添加OTA更新能力
低功耗模式管理
```

### 算法层
```
RandomForest (现) → Conv1D+Bi-LSTM (升级)
特征维度: 11维 (保持)
输入格式: 单帧 → 100帧时间序列
模型大小: ~200-300KB (可放入ESP32)
```

### 应用层
```
单向输出 → 置信度+结果返回
TTS反馈 → 多模式反馈(LED/震动/TTS)
手机App → 数据可视化+模型管理
```
